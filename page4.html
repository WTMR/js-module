jQuery插件支持AMD

模块定义
(function(factory) {
  if (typeof define === "function" && define.amd) {
    define(["jquery"], factory);
  } else {
    factory(jQuery);
  }
}(function($) {
  
  $.extend({
    say: function() {
      alert("jquery say")
    }
  })

}));
config.js配置
require.config({　　　　
  baseUrl: "./assets/js",
  paths: {　　　　　　
    "jquery": "jquery.min",
    "jquery-say": "jquery.say"
  }
})
模块调用
requirejs(["jquery","jquery-say"], function() {
   $.say();
});
